Example pipeline for '**custom.xml'

  <map:pipeline>
   <map:match pattern="**custom.xml">
    <map:generate src="{project:content.xdocs}{1}custom.xml" />
    <map:transform src="{project:resources.stylesheets}/hello2document.xsl" />
    <map:serialize type="xml"/>
   </map:match>
  </map:pipeline>
 </map:pipelines>
</map:sitemap>
